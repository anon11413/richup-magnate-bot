    // Record current sell prices for all owned properties (called each game turn)
    function recordPriceSnapshot() {
        if (!state.pid || state.blocks.length === 0) return;
        const turn = state.gameTurn;
        state.blocks.forEach((b, idx) => {
            if (b.ownerId === state.pid && b.type === 'city') {
                const price = getSellPrice(idx);
                if (!state.priceHistory[idx]) {
                    state.priceHistory[idx] = [];
                }
                const hist = state.priceHistory[idx];
                // Only record one entry per turn
                if (hist.length === 0 || hist[hist.length - 1].turn !== turn) {
                    hist.push({ turn: turn, p: price });
                } else {
                    // Same turn — update price (may have changed within the turn)
                    hist[hist.length - 1].p = price;
                }
                // No cap — show full history from game start
            }
        });
    }

    // Record lobby net worth snapshot (called each game turn)
    // V18.4: Record inflation snapshot with cash offset applied.
    // The offset compensates for bankrupt players' cash leaving the economy
    // (artificial, not real deflation) while property losses show as real dips.
    function recordInflationSnapshot() {
        const turn = state.gameTurn;
        const hist = state.inflationHistory;
        if (hist.length === 0 || hist[hist.length - 1].turn !== turn) {
            hist.push({ turn: turn, netWorth: state.lobbyNet });
        } else {
            hist[hist.length - 1].netWorth = state.lobbyNet;
        }
    }

    function wsSend(eventName, payload) {
        if (!state.ws) { addLog("NO WS CONNECTION", "#ff4757"); return false; }
        try {
            const msg = payload !== undefined
                ? `42/api/game,${JSON.stringify([eventName, payload])}`
                : `42/api/game,${JSON.stringify([eventName])}`;
            state.ws.send(msg);
            addLog(`TX: ${eventName}`, "#3498db");

    // ============================================================
    // 3. WEBSOCKET HIJACK (IRONCLAD - works at any injection timing)
    // ============================================================
    let _wsAttachCount = 0;       // Track how many times we've attached (for reconnection)
    const _attachedWsSeen = new WeakSet(); // Prevent duplicate listeners on same WS object
    let _wsScanTimer = null;      // Scan interval reference (also stored as window.__magnateWsScan)
    let _wsScanRate = 500;        // Start fast, slow down once connected
    let _wsScanAttempts = 0;

    function attachToWS(ws, source) {
        // Allow re-attach if previous WS is dead
        if (state.ws && state.ws.readyState === 1) return;
        // Prevent duplicate listeners on the same WS object
        if (_attachedWsSeen.has(ws)) {
            state.ws = ws; // Update reference but don't re-add listeners
            return;
        }
        _attachedWsSeen.add(ws);
        if (state.ws && state.ws !== ws) {
            addLog(`WS RECONNECT (old was ${state.ws.readyState})`, "#f39c12");
        }

        state.ws = ws;
        _wsAttachCount++;
        document.getElementById('mag-ws').innerText = 'ONLINE';
        document.getElementById('mag-ws').style.color = '#2ecc71';
        addLog(`WS HIJACKED #${_wsAttachCount} (${source})`, "#2ecc71");

        // Listen for messages
        ws.addEventListener('message', (e) => handleMessage(e.data));

        // CRITICAL: Listen for close/error to auto-reconnect
        ws.addEventListener('close', () => {
            addLog("WS CLOSED - scanning for new connection...", "#ff4757");
            document.getElementById('mag-ws').innerText = 'RECONNECTING';
            document.getElementById('mag-ws').style.color = '#f39c12';
            state.ws = null;
            startWSScan(); // Restart scanning
        });
        ws.addEventListener('error', () => {
            addLog("WS ERROR - will reconnect on close", "#ff4757");
        });

        // Request full game state from server (critical for mid-game injection)
        // Multiple attempts at increasing intervals to handle various timings
        const syncDelays = [500, 2000, 5000, 10000, 20000];
        syncDelays.forEach((delay, i) => {
            setTimeout(() => {
                if (state.ws !== ws) return; // WS changed, stop
                if (i > 0 && state.pid && state.blocks.length > 0) return; // Already synced
                wsSend("request-sync", { snapshot: "0" });
                if (i === 0) addLog("Requested game state sync...", "#f39c12");
                else addLog(`Sync retry #${i + 1}...`, "#f39c12");
            }, delay);
        });

        // DOM-based state recovery at multiple intervals
        [1000, 3000, 6000, 12000].forEach(delay => {
            setTimeout(() => {
                if (state.ws !== ws) return;
                syncStateFromDOM();
            }, delay);
        });

        // Slow down the scanner now that we're connected
        if (_wsScanTimer) { clearInterval(_wsScanTimer); _wsScanTimer = null; }
    }

    // Detect game state from DOM when injected mid-game
    let _lastDomRollTime = 0; // Prevent double-roll from DOM scan + WS event overlap

    function syncStateFromDOM() {
        if (!state.ws) return;

        // Check for Roll button (indicates it's our turn)
        const rollBtn = Array.from(document.querySelectorAll('button, [role="button"]')).find(el => {
            const text = (el.innerText || el.getAttribute('aria-label') || '').toLowerCase();
            return text.includes('roll') && el.offsetWidth > 0 && !el.closest('#mag-v17');
        });
        if (rollBtn && !isInsideTradeUI(rollBtn)) {
            // Only act if we haven't already detected our turn and haven't rolled yet
            if (!state.isMyTurn) {
                addLog("Detected ROLL button - it's our turn!", "#f1c40f");
                state.isMyTurn = true;
                state.diceRolled = false;
            }
            // Guard: don't send roll if dice already rolled or if we rolled recently (prevents double-roll)
            const now = Date.now();
            if (state.enabled && isAutoRoll() && !state.diceRolled && (now - _lastDomRollTime > 3000)) {
                _lastDomRollTime = now;
                wsSend("plz-roll-dices");
                state.stats.rolls++;
                updateStatsUI();
            }
        }

        // Try to recover block/player state from React fiber tree
        if (state.blocks.length === 0) {
            recoverStateFromReactFiber();
        }

        // Try to detect our identity from the page if we don't have it
        if (!state.pid) {
            recoverIdentityFromDOM();
        }
    }

    // Crawl React fiber tree to find game state (board blocks, players)
    function recoverStateFromReactFiber() {
        try {
            const appRoot = document.getElementById('root') || document.getElementById('app') || document.querySelector('[data-reactroot]');
            if (!appRoot) return;
            const fiberKey = Object.keys(appRoot).find(k => k.startsWith('__reactFiber$') || k.startsWith('__reactInternalInstance$'));
            if (!fiberKey) return;
            let fiber = appRoot[fiberKey];
            let depth = 0;
            const visited = new Set();
            while (fiber && depth < 50) {
                if (visited.has(fiber)) break;
                visited.add(fiber);
                const s = fiber.memoizedState;
                const p = fiber.memoizedProps;
                // Look for game state in memoizedState/props
                if (s && s.memoizedState && Array.isArray(s.memoizedState.blocks)) {
                    state.blocks = s.memoizedState.blocks;
                    addLog(`Recovered ${state.blocks.length} blocks from fiber!`, "#2ecc71");
                }
                if (p && Array.isArray(p.blocks) && p.blocks.length > 10) {
                    state.blocks = p.blocks;
                    addLog(`Recovered ${state.blocks.length} blocks from fiber props!`, "#2ecc71");
                }
                if (p && p.participants && Array.isArray(p.participants)) {
                    p.participants.forEach(pl => { if (pl.id) state.players[pl.id] = pl; });
                }
                fiber = fiber.child || fiber.sibling || fiber.return;
                depth++;
            }
        } catch (e) { /* fiber crawl failed, not critical */ }
    }

    // Try to find our identity from DOM elements showing player name
    function recoverIdentityFromDOM() {
        if (state.pid) return;
        // Look for elements that contain our target name
        try {
            const allText = document.querySelectorAll('span, div, p');
            for (const el of allText) {
                if (el.offsetWidth === 0) continue;
                if (el.closest('#mag-v17') || el.closest('#mag-props-panel')) continue;
                const text = (el.innerText || '').trim();
                if (text.toLowerCase() === CONFIG.targetName.toLowerCase()) {
                    // Found our name on screen - check if there's player data in fiber
                    const fiberKey = Object.keys(el).find(k => k.startsWith('__reactFiber$'));
                    if (fiberKey) {
                        let fiber = el[fiberKey];
                        let d = 0;
                        while (fiber && d < 15) {
                            const props = fiber.memoizedProps;
                            if (props && props.player && props.player.id) {
                                lockIdentity(props.player);
                                return;
                            }
                            fiber = fiber.return;
                            d++;
                        }
                    }
                }
            }
        } catch(e) {}
    }

    // ---- METHOD 1: WebSocket Constructor Proxy (catches ALL future connections) ----
    const _OriginalWS = window.WebSocket;
    const _OriginalSend = _OriginalWS.prototype.send;
    window.WebSocket = function(...args) {

            });
        }

        window.activeWS = ws;
        return ws;
    };

    // ---- METHOD 4: Aggressive repeating scanner with adaptive rate ----
    function deepScanForWS() {
        // Check known globals
        const knownKeys = ['activeWS', '_ws', 'socket', 'ws', 'gameSocket', 'io'];
        for (const key of knownKeys) {
            try {
                const val = window[key];
                if (val instanceof _OriginalWS && val.readyState === 1) {
                    attachToWS(val, 'global-' + key);
                    return true;
                }
            } catch(e) {}
        }

        // Deep scan: search ALL window properties for WebSocket instances (3 levels deep)
        try {
            for (const key of Object.getOwnPropertyNames(window)) {
                try {
                    const val = window[key];
                    if (val instanceof _OriginalWS && val.readyState === 1) {
                        attachToWS(val, 'deep-' + key);
                        return true;
                    }
                    if (val && typeof val === 'object' && !Array.isArray(val)) {
                        for (const k2 of Object.keys(val)) {
                            try {
                                const v2 = val[k2];
                                if (v2 instanceof _OriginalWS && v2.readyState === 1) {
                                    attachToWS(v2, 'deep-' + key + '.' + k2);
                                    return true;
                                }
                                // Socket.IO nests deeply: manager.engine.transport.ws
                                if (v2 && typeof v2 === 'object') {
                                    for (const k3 of Object.keys(v2)) {
                                        try {
                                            const v3 = v2[k3];
                                            if (v3 instanceof _OriginalWS && v3.readyState === 1) {
                                                attachToWS(v3, 'deep-' + key + '.' + k2 + '.' + k3);
                                                return true;
                                            }
                                            if (v3 && typeof v3 === 'object') {
                                                for (const k4 of Object.keys(v3)) {
                                                    try {
                                                        const v4 = v3[k4];
                                                        if (v4 instanceof _OriginalWS && v4.readyState === 1) {
                                                            attachToWS(v4, 'deep-' + key + '.' + k2 + '.' + k3 + '.' + k4);
                                                            return true;
                                                        }
                                                    } catch(e) {}
                                                }
                                            }
                                        } catch(e) {}
                                    }
                                }
                            } catch(e) {}
                        }
                    }
                } catch (e) {}
            }
        } catch (e) {}

        // Socket.IO specific: look for io.Manager instances
        try {
            if (window.io && window.io.managers) {
                for (const url of Object.keys(window.io.managers)) {
                    const mgr = window.io.managers[url];
                    if (mgr && mgr.engine && mgr.engine.transport && mgr.engine.transport.ws) {
                        const ws = mgr.engine.transport.ws;
                        if (ws instanceof _OriginalWS && ws.readyState === 1) {
                            attachToWS(ws, 'io-manager');
                            return true;
                        }
                    }
                }
            }
        } catch(e) {}

        return false;
    }

    function startWSScan() {
        if (_wsScanTimer) clearInterval(_wsScanTimer);
        _wsScanRate = 500; // Start fast
        _wsScanAttempts = 0;

        function scanTick() {
            _wsScanAttempts++;

            // If we already have a live WS, stop scanning
            if (state.ws && state.ws.readyState === 1) {
                clearInterval(_wsScanTimer);
                _wsScanTimer = null;
                return;
            }

            // Clear dead WS reference
            if (state.ws && state.ws.readyState !== 1) {
                state.ws = null;
            }

            const found = deepScanForWS();
            if (found) {
                clearInterval(_wsScanTimer);
                _wsScanTimer = null;
                return;
            }

            // Adaptive rate: speed up initially, then slow down
            // 500ms for first 20 attempts (10s), then 2s for next 30 (60s), then 5s forever
            if (_wsScanAttempts === 20 && _wsScanRate === 500) {
                clearInterval(_wsScanTimer);
                _wsScanRate = 2000;
                _wsScanTimer = setInterval(scanTick, _wsScanRate);
                addLog("WS scan: slowing to 2s interval...", "#aaa");
            } else if (_wsScanAttempts === 50 && _wsScanRate === 2000) {
                clearInterval(_wsScanTimer);
                _wsScanRate = 5000;
                _wsScanTimer = setInterval(scanTick, _wsScanRate);
                addLog("WS scan: slowing to 5s interval...", "#aaa");
            }

            // Log progress periodically
            if (_wsScanAttempts % 10 === 0) {
                addLog(`WS scan: ${_wsScanAttempts} attempts, waiting for connection...`, "#7f8c8d");
            }
        }

        _wsScanTimer = setInterval(scanTick, _wsScanRate);
        window.__magnateWsScan = _wsScanTimer;
    }

    // 4. EVENT-DRIVEN PROTOCOL HANDLER (THE BRAIN)

    function handleMessage(payload) {
        if (typeof payload !== 'string') return;
        if (!payload.startsWith('42/api/game,')) return;

        try {
            const raw = payload.substring(payload.indexOf('['));
            const parsed = JSON.parse(raw);
            const ev = parsed[0];
            const body = parsed[1] || {};

            switch (ev) {
                case 'entered-room':     handleEnteredRoom(body); break;
                case 'joined-game':      handleJoinedGame(body); break;
                case 'player-joined':    handlePlayerJoined(body); break;
                case 'game-started':     handleGameStarted(body); break;
                case 'plz-roll-dices':   handlePlzRollDices(); break;
                case 'dice-rolled':      handleDiceRolled(body); break;
                case 'purchase-success': handlePurchaseSuccess(body); break;
                case 'turn-ended':       handleTurnEnded(); break;
                case 'money-changed':    handleMoneyChanged(body); break;
                case 'sync-game-state':  handleSyncGameState(body); break;
                case 'trade-created':    handleTradeCreated(body); break;
                case 'trade-declined':   handleTradeDeclined(body); break;
                case 'trade-confirmed':  handleTradeConfirmed(body); break;
                case 'game-error':       handleGameError(body); break;
                case 'game-room-updated':
                    if (body.settings) addLog("Settings updated", "#aaa");
                    break;
                case 'city-level-changed':
                    // Track house/hotel building (affects net worth calculation)
                    if (body.blockIndex !== undefined && state.blocks[body.blockIndex]) {
                        state.blocks[body.blockIndex].level = body.level || 0;
                        addLog(`Building: #${body.blockIndex} now level ${body.level}`, "#aaa");
                        sendInstantUpdate();
                    }
                    break;
                case 'property-mortgaged':
                    if (body.blockIndex !== undefined && state.blocks[body.blockIndex]) {
                        state.blocks[body.blockIndex].isMortgaged = true;
                        addLog(`Mortgaged: ${state.blocks[body.blockIndex].name || '#' + body.blockIndex}`, "#aaa");
                        sendInstantUpdate();
                    }
                    break;
                case 'property-unmortgaged':
                    if (body.blockIndex !== undefined && state.blocks[body.blockIndex]) {
                        state.blocks[body.blockIndex].isMortgaged = false;
                        addLog(`Unmortgaged: ${state.blocks[body.blockIndex].name || '#' + body.blockIndex}`, "#aaa");
                        sendInstantUpdate();
                    }
                    break;
                case 'paid-out-of-prison':
                    addLog("Paid out of prison", "#f39c12");
                    break;
                // V18.1 FIX: Handle player removal events to keep lobbyNet accurate
                case 'player-left':
                case 'player-bankrupt':
                case 'player-eliminated':
                case 'player-disconnected':
                case 'player-removed':
                    handlePlayerRemoved(body, ev);
                    break;
                default:
                    // Catch property-bought events (broadcast when others buy)
                    if (ev === 'property-bought' || (typeof ev === 'string' && ev.includes('bought'))) {
                        handleOtherPlayerBought(body);
                    }
                    // V18.1 FIX: Catch any other player-departure event names we didn't list
                    if (typeof ev === 'string' && (ev.includes('bankrupt') || ev.includes('eliminated') || ev.includes('player-left') || ev.includes('player-removed'))) {
                        handlePlayerRemoved(body, ev);
                    }
                    break;
            }

            // Identity scan on all messages
            scanForIdentity(body);

        } catch (e) { /* parse error, ignore non-game messages */ }
    }

    // --- Room & Identity ---
    function handleEnteredRoom(body) {
        state.blocks = body.room?.blocks || state.blocks;
        state.currentPlayerIndex = body.room?.currentPlayerIndex || 0;
        (body.room?.participants || []).forEach(p => { state.players[p.id] = p; });
        addLog(`Entered room: ${state.blocks.length} blocks, ${Object.keys(state.players).length} players`, "#2ecc71");
    }

    function handleJoinedGame(body) {
        if (body.selfPlayer) {
            // Always update player data, but only set pid if not already locked
            // (LINK IDENTITY manual override takes priority)
            if (!state.pid) {
                state.pid = body.selfPlayer.id;
                updateMeUI(body.selfPlayer.name);
                addLog(`Identity: ${body.selfPlayer.name} (${state.pid.slice(0,8)})`, "#2ecc71");
            }
            state.players[body.selfPlayer.id] = body.selfPlayer;
            if (body.selfPlayer.id === state.pid) {
                state.myPosition = body.selfPlayer.position || 0;
                state.myMoney = (typeof body.selfPlayer.money === 'number') ? body.selfPlayer.money : 1500;
                updateStatsUI();
            }
        }
    }

    function handlePlayerJoined(body) {
        if (body.id) state.players[body.id] = body;

    }

    function handleGameStarted(body) {

        const myIdx = state.participantsOrder.findIndex(p => p.id === state.pid);
        if (myIdx >= 0) {
            addLog(`We are player #${myIdx + 1} in order`, "#2ecc71");
            if (myIdx === 0) {
                addLog("We're FIRST - auto-rolling!", "#f1c40f");
                setTimeout(() => handlePlzRollDices(), 1500);
            }
        }
        // Mortgaging happens on our turn (see handleDiceRolled)

        // Record initial state at turn 0
        state.gameTurn = 0;
        state.turnEndedCount = 0;
        state.inflationHistory = [];
        state.priceHistory = {};
        // V18.5: No offset — clean raw tracking

        }, 2000);

    // --- Full state sync (periodic from server) ---
    function handleSyncGameState(body) {
        const gs = body.gameState;
        if (!gs) return;
        state.blocks = gs.blocks || state.blocks;
        state.currentPlayerIndex = gs.currentPlayerIndex || 0;
        if (gs.participantsOrder) {
            state.participantsOrder = gs.participantsOrder;
        }

        const participants = gs.participants || gs.participantsOrder || [];
        // Replace players map with fresh data to remove bankrupt/disconnected players
        const freshPlayers = {};
        participants.forEach(p => {
            freshPlayers[p.id] = p;
            if (p.id === state.pid) {
                state.myPosition = p.position || 0;
                state.myMoney = (typeof p.money === 'number') ? p.money : 0;
                if (p.debtTo) {
                    addLog(`DEBT detected to ${p.debtTo}!`, "#ff4757");
                    tryMortgage();
                }
            }
        });
        // Replace state.players with fresh data (removes bankrupt/disconnected players)
        state.players = freshPlayers;

        // Identity detection: if we don't have pid yet, find ourselves
        if (!state.pid && participants.length > 0) {
            // Method 1: Match by targetName
            const byName = participants.find(p =>
                p.name && p.name.toLowerCase().includes(CONFIG.targetName.toLowerCase()));
            if (byName) {
                lockIdentity(byName);
            } else {
                // Method 2: Only non-bot human player
                const humans = participants.filter(p => !p.isBot);
                if (humans.length === 1) {
                    lockIdentity(humans[0]);
                } else if (humans.length > 1) {
                    addLog(`${humans.length} humans found. Use LINK IDENTITY.`, "#ff4757");
                }
            }
        }

        // Update our money from players if we have identity
        if (state.pid && state.players[state.pid]) {
            state.myMoney = state.players[state.pid].money || 0;
            state.myPosition = state.players[state.pid].position || 0;
        }

        addLog(`Synced: ${participants.length} players, ${state.blocks.length} blocks`, "#2ecc71");
        updateStatsUI();
        // Mortgage on sync only if it's our turn (skip if already mortgaging)
        if (state.isMyTurn && state.enabled && isAutoMortgage() && state.mortgagingDone) {
            state.mortgagingDone = false;
            // Safety timeout: force mortgagingDone=true after 30s to prevent stuck turns
            const _mortSafe1 = setTimeout(() => { state.mortgagingDone = true; }, 30000);
            setTimeout(() => {
                mortgageAllOnTurn(() => { state.mortgagingDone = true; clearTimeout(_mortSafe1); });
            }, 1000);
        }
    }

    function handleDiceRolled(body) {
        state.diceRolled = true;
        state.purchaseSent = false;
        state.lastDice = body.dice || [0, 0];
        const total = state.lastDice[0] + state.lastDice[1];
        const isDoubles = state.lastDice[0] === state.lastDice[1] && state.lastDice[0] > 0;
        state.myPosition = (state.myPosition + total) % (state.blocks.length || 40);

        addLog(`Dice: [${state.lastDice}] = ${total} -> pos ${state.myPosition}${isDoubles ? ' DOUBLES!' : ''}`, isDoubles ? "#f1c40f" : "#3498db");
        updateStatsUI();

        // Auto-buy: send purchase-property, let server reject if not applicable
        if (state.isMyTurn && state.enabled && isAutoBuy()) {
            setTimeout(() => {
                if (!state.purchaseSent) {
                    state.purchaseSent = true;
                    wsSend("purchase-property");
                }
            }, 500 + Math.random() * 500);
        }

        // Click safe game buttons as fallback (Buy/Pass/Pay/OK)
        setTimeout(() => clickSafeButtons(), 1500);
        setTimeout(() => clickSafeButtons(), 3000);

        // After purchase attempt, mortgage all unmortgaged properties via UI clicks
        if (state.isMyTurn && state.enabled && isAutoMortgage()) {
            state.mortgagingDone = false;
            // Safety timeout: force mortgagingDone=true after 30s to prevent stuck turns
            const _mortSafe2 = setTimeout(() => { state.mortgagingDone = true; addLog("Mortgage safety timeout - unblocking End Turn", "#ff4757"); }, 30000);
            setTimeout(() => {
                mortgageAllOnTurn(() => {
                    state.mortgagingDone = true;
                    clearTimeout(_mortSafe2);
                    addLog("Mortgaging complete", "#2ecc71");
                    // If doubles, re-roll now that mortgaging is done
                    if (isDoubles && state.isMyTurn && state.enabled && isAutoRoll()) {
                        setTimeout(() => {
                            if (state.isMyTurn && state.diceRolled) {
                                state.diceRolled = false;
                                state.purchaseSent = false;
                                addLog("Re-rolling for DOUBLES...", "#f1c40f");
                                wsSend("plz-roll-dices");
                                state.stats.rolls++;
                                updateStatsUI();
                            }
                        }, 500);
                    }
                });
            }, 2000);
        } else if (isDoubles && state.isMyTurn && state.enabled && isAutoRoll()) {
            // DOUBLES without mortgage: re-roll after a short delay
            addLog("DOUBLES detected - will re-roll!", "#f1c40f");
            setTimeout(() => {
                if (state.isMyTurn && state.diceRolled) {
                    state.diceRolled = false;
                    state.purchaseSent = false;
                    addLog("Re-rolling for DOUBLES...", "#f1c40f");
                    wsSend("plz-roll-dices");
                    state.stats.rolls++;
                    updateStatsUI();

    function handlePurchaseSuccess(body) {
        const bIdx = body.blockIndex;
        const currentPlayer = getCurrentPlayerId();
        // Determine who bought: if it's our turn, it's us. Otherwise it's the current player.
        const isOurPurchase = state.isMyTurn || currentPlayer === state.pid || !currentPlayer;
        const buyerId = isOurPurchase ? state.pid : currentPlayer;

        if (state.blocks[bIdx]) {
            state.blocks[bIdx].ownerId = buyerId;
            // V18.5 FIX: Deduct purchase price from buyer's cash locally.
            // Server only sends money-changed to the affected player, not broadcast.
            // Without this, net worth inflates because we add property value but never deduct cash.
            if (buyerId && state.players[buyerId] && state.blocks[bIdx].price) {
                state.players[buyerId].money = (state.players[buyerId].money || 0) - state.blocks[bIdx].price;
            }
        }

        if (isOurPurchase) {
            state.stats.buys++;
            addLog(`BOUGHT: ${state.blocks[bIdx]?.name || '#' + bIdx}`, "#2ecc71");
            updateStatsUI();
        } else {
            const name = state.players[buyerId]?.name || '?';
            addLog(`${name} bought ${state.blocks[bIdx]?.name || '#' + bIdx}`, "#aaa");
        }
        sendInstantUpdate();  // Push immediately
    }

    function handleTurnEnded() {
        const wasMyTurn = state.isMyTurn;
        state.isMyTurn = false;
        state.diceRolled = false;
        state.purchaseSent = false;

        const prevIdx = state.currentPlayerIndex;
        state.currentPlayerIndex = (state.currentPlayerIndex + 1) % (state.participantsOrder.length || 4);

        // Track individual turn-ended events for game turn calculation
        state.turnEndedCount++;

        // Detect new full game turn (all players went once)
        // When currentPlayerIndex wraps back to 0, one full round completed
        if (state.currentPlayerIndex === 0 && prevIdx > 0 && state.participantsOrder.length > 0) {
            state.gameTurn++;
            const gtEl = document.getElementById('mag-gameturn');
            if (gtEl) gtEl.innerText = state.gameTurn;
            addLog(`GAME TURN ${state.gameTurn} complete`, "#9b59b6");
            // Record snapshots at each game turn
            recalcLobbyNet();  // Ensure fresh net worth
            recordPriceSnapshot();
            recordInflationSnapshot();
            sendLiveSnapshot();  // V18.5: Push to live tracker
        }

        document.getElementById('mag-turn').innerText = 'WAITING';
        document.getElementById('mag-turn').style.color = '#aaa';

        if (wasMyTurn) {
            state.stats.turns++;
            // V18.5 FIX: Use our personal turn count as game turn since
            // turn-ended only fires for our turns (bots resolve server-side).
            // Each of our turns = 1 full game round (all bots went between our turns).
            if (state.stats.turns > state.gameTurn) {
                state.gameTurn = state.stats.turns;
                const gtEl = document.getElementById('mag-gameturn');
                if (gtEl) gtEl.innerText = state.gameTurn;
                addLog(`GAME TURN ${state.gameTurn} complete`, "#9b59b6");
                recalcLobbyNet();
                recordPriceSnapshot();
                recordInflationSnapshot();
                sendLiveSnapshot();
            }
            updateStatsUI();
            addLog(`My turn complete (#${state.stats.turns})`, "#aaa");
        }
    }

    function handleMoneyChanged(body) {
        if (state.players[body.playerId]) {

    }

    function handleGameError(body) {

    // V18.1 FIX: Handle player leaving/bankrupt/eliminated
    // Removes the player from state.players so their cash no longer inflates lobbyNet.
    // Clears ownerId on their properties so those don't inflate lobbyNet either.
    // Cleans up any active negotiations targeting the gone player.
    function handlePlayerRemoved(body, eventName) {
        const removedId = body.playerId || body.id;
        if (!removedId) {
            // Some events send the player object directly
            if (body.player && body.player.id) {
                handlePlayerRemoved({ playerId: body.player.id }, eventName);
            }
            return;
        }

        const playerName = state.players[removedId]?.name || removedId.slice(0, 8);
        addLog(`PLAYER GONE: ${playerName} (${eventName})`, "#ff4757");

        // V18.5: Clean raw tracking — no offset. Bankruptcy shows as a real dip
        // (player's cash + properties leave the economy). Chart recovers naturally
        // as properties get re-bought by remaining players.

        // Remove from players map
        delete state.players[removedId];

        // Clear ownerId on all their properties — value leaves the economy
        state.blocks.forEach((b, idx) => {
            if (b.ownerId === removedId) {
                b.ownerId = null;
            }
        });

        // Remove from participantsOrder (affects turn tracking)
        if (state.participantsOrder.length > 0) {
            const oldLen = state.participantsOrder.length;
            const removedIdx = state.participantsOrder.findIndex(p => p.id === removedId);
            if (removedIdx >= 0) {
                state.participantsOrder.splice(removedIdx, 1);
                // Adjust currentPlayerIndex if needed
                if (removedIdx < state.currentPlayerIndex) {
                    state.currentPlayerIndex = Math.max(0, state.currentPlayerIndex - 1);
                } else if (state.currentPlayerIndex >= state.participantsOrder.length) {
                    state.currentPlayerIndex = 0;
                }
            }
        }

        // Cancel any negotiations targeting the gone player
        for (const bIdx in state.negs) {
            if (state.negs[bIdx].target === removedId) {
                addLog(`Cancelled negotiation for #${bIdx} (player gone)`, "#aaa");
                delete state.negs[bIdx];
            }
        }

        // Force immediate lobbyNet recalculation so the dashboard updates right away
        recalcLobbyNet();

        // V18.4: Record inflation snapshot immediately after bankruptcy
        // so the property deflation shows on the chart at the right moment
        recordInflationSnapshot();

        // V18.5: Push bankruptcy event to live tracker
        sendLiveBankruptcy(playerName);

        // Also request a fresh sync from server to stay accurate
        if (state.ws) {
            setTimeout(() => wsSend("request-sync", { snapshot: "0" }), 2000);
        }
    }

                state.stats.declined++;
                addLog(`AUTO-DECLINED trade ${trade.id.slice(0,8)}`, "#ff4757");
            }, 500 + Math.random() * 1000);
        }

        // If WE are the initiator, track it
        if (trade.initiatorId === state.pid) {

        }
    }

    function handleTradeConfirmed(body) {
        addLog(`Trade confirmed: ${body.tradeId?.slice(0,8)}`, "#2ecc71");
        // Update block ownership from trade data (critical for accurate net worth)
        if (body.trade) {
            const t = body.trade;
            // Properties moving from initiator to recipient
            if (t.myOffer?.blockIndices) {
                t.myOffer.blockIndices.forEach(idx => {
                    if (state.blocks[idx]) state.blocks[idx].ownerId = t.recipientId;
                });
            }
            // Properties moving from recipient to initiator
            if (t.theirOffer?.blockIndices) {
                t.theirOffer.blockIndices.forEach(idx => {
                    if (state.blocks[idx]) state.blocks[idx].ownerId = t.initiatorId;
                });
            }
        }
        // Request fresh sync to ensure accuracy
        setTimeout(() => { wsSend("request-sync", { snapshot: "0" }); sendInstantUpdate(); }, 1000);
    }

    // Get color sets we own at least 1 property from
    function getMyColorSets() {
        const sets = new Set();
        state.blocks.forEach(b => {
            if (b.type === 'city' && b.ownerId === state.pid && b.countryId) {
                sets.add(b.countryId);
            }

        return sets;

    // Check if a block is a valid auto-buy target
    function isValidBuyTarget(blockIndex) {
        const block = state.blocks[blockIndex];
        if (!block) return false;
        // Must be a city (not airport, company, corner, bonus)
        if (block.type !== 'city') return false;
        // Must be owned by someone else
        if (!block.ownerId || block.ownerId === state.pid) return false;
        // Must NOT be from a color set we already own a piece of
        const mySets = getMyColorSets();
        if (mySets.has(block.countryId)) return false;
        return true;
    }

    // Triggered when another player buys a property
    function handleOtherPlayerBought(body) {
        const bIdx = body.blockIndex;
        const pId = body.playerId || getCurrentPlayerId();
        if (bIdx === undefined || !pId || pId === state.pid) return;

        // Update block ownership + deduct cash locally
        if (state.blocks[bIdx]) {
            state.blocks[bIdx].ownerId = pId;
            if (pId && state.players[pId] && state.blocks[bIdx].price) {
                state.players[pId].money = (state.players[pId].money || 0) - state.blocks[bIdx].price;
            }
        }

        // Check if this is a valid negotiation target
        if (!isValidBuyTarget(bIdx)) return;

        // Don't duplicate existing negotiation
        if (state.negs[bIdx]) return;

        const block = state.blocks[bIdx];
        const playerName = state.players[pId]?.name || pId.slice(0,8);
        addLog(`TARGET: ${block.name} (${block.countryId}) owned by ${playerName}`, "#f39c12");
        state.negs[bIdx] = {
            stage: 0,
            time: 0,            // Will fire on first check
            target: pId,
            cooldownUntil: 0
        };
    }

    // Periodic scan: find all blocks owned by others that qualify for negotiation
    function scanForNegotiationTargets() {
        if (!state.pid || state.blocks.length === 0) return;
        state.blocks.forEach((block, idx) => {
            if (!isValidBuyTarget(idx)) return;
            if (state.negs[idx]) return; // Already tracking
            const playerName = state.players[block.ownerId]?.name || block.ownerId?.slice(0,8) || '?';
            addLog(`TARGET: ${block.name} (${block.countryId}) owned by ${playerName}`, "#f39c12");
            state.negs[idx] = {
                stage: 0,
                time: 0,
                target: block.ownerId,
                cooldownUntil: 0
            };
        });
    }

    // Get current player based on turn index
    function getCurrentPlayerId() {
        if (state.participantsOrder.length > 0 && state.currentPlayerIndex < state.participantsOrder.length) {
            return state.participantsOrder[state.currentPlayerIndex].id;
        }
        return null;
    }

    // Run negotiation cycle (called from main loop)
    function runNegotiations() {
        if (!state.ws || !state.pid) return;
        const now = Date.now();

        for (const bIdx in state.negs) {
            const n = state.negs[bIdx];

            // Re-validate: if we now own a property in this color set, cancel negotiation
            if (!isValidBuyTarget(parseInt(bIdx))) {
                addLog(`Cancelling neg for #${bIdx} (no longer valid)`, "#aaa");
                delete state.negs[bIdx];
                continue;
            }

            // Check if in cooldown
            if (n.cooldownUntil > 0 && now < n.cooldownUntil) continue;

            // Reset after cooldown expires
            if (n.cooldownUntil > 0 && now >= n.cooldownUntil) {
                n.stage = 0;
                n.cooldownUntil = 0;
                addLog(`Cooldown over for #${bIdx}, restarting offers`, "#f39c12");
            }

            // All offers exhausted? Start cooldown
            if (n.stage >= CONFIG.offers.length) {
                n.cooldownUntil = now + CONFIG.offerCooldown;
                addLog(`All offers exhausted for #${bIdx}, 5min cooldown`, "#aaa");
                continue;
            }

            // Time for next offer?
            if (now - n.time < CONFIG.offerInterval) continue;

            // Cash check: don't offer more than we have
            const amt = CONFIG.offers[n.stage];
            if (state.myMoney < amt) {
                addLog(`Skip offer $${amt} for #${bIdx} (only $${state.myMoney})`, "#ff4757");
                continue; // Don't advance stage, just wait until we have money
            }

            // Send trade offer
            const block = state.blocks[parseInt(bIdx)];
            const targetName = state.players[n.target]?.name || '?';
            wsSend("create-trade", {
                targetPlayerId: n.target,
                initiatorOffer: { money: amt, propertyIndices: [], mortgagedPropertiesIndexes: [], pardonCards: [] },
                recipientOffer: { money: 0, propertyIndices: [parseInt(bIdx)], mortgagedPropertiesIndexes: [], pardonCards: [] }
            });
            n.time = now;
            n.stage++;
            state.stats.offered++;
            addLog(`OFFER: $${amt} to ${targetName} for ${block?.name || '#'+bIdx}`, "#3498db");
        }
    }

    // Find all board tile DOM elements for properties we own that aren't mortgaged
    function findOwnedUnmortgagedTiles() {
        const tiles = [];
        const seen = new Set();
        const allEls = document.querySelectorAll('div, span');

        for (const el of allEls) {
            if (el.offsetWidth === 0 || el.offsetHeight === 0) continue;
            const rect = el.getBoundingClientRect();
            if (rect.width > 200 || rect.height > 200 || rect.width < 20 || rect.height < 20) continue;

            const fiberKey = Object.keys(el).find(k => k.startsWith('__reactFiber$'));
            if (!fiberKey) continue;

            let fiber = el[fiberKey];
            let depth = 0;
            while (fiber && depth < 12) {
                const block = fiber.memoizedProps?.block;
                if (block && block.ownerId === state.pid && !block.isMortgaged && block.price && !seen.has(block.name)) {
                    seen.add(block.name);
                    tiles.push({ el, name: block.name, price: block.price });
                    break;
                }
                fiber = fiber.return;
                depth++;
            }
        }
        return tiles;
    }

    // Mortgage one property via UI: click tile → wait → click Mortgage button → wait → close
    function mortgageOneViaUI(tileInfo, callback) {
        if (!state.isMyTurn) { callback(); return; }

        // Step 1: Click the tile to open property popup
        addLog(`Clicking tile: ${tileInfo.name}`, "#f39c12");
        try {
            tileInfo.el.click(); // Real DOM click to trigger tippy
        } catch (e) {
            addLog(`Tile click failed: ${e.message}`, "#ff4757");
            callback();
            return;
        }

        // Step 2: Wait for popup, find and click mortgage button
        setTimeout(() => {
            const btn = findMortgageButton();
            if (btn) {
                addLog(`MORTGAGE: ${tileInfo.name} +$${Math.floor((tileInfo.price || 0) / 2)}`, "#2ecc71");
                reactClick(btn);
                state.stats.mortgaged++;
                // Mark as mortgaged locally to prevent double-counting in lobbyNet
                const bIdx = state.blocks.findIndex(b => b.name === tileInfo.name && b.ownerId === state.pid);
                if (bIdx >= 0) state.blocks[bIdx].isMortgaged = true;

                // Step 3: Wait for mortgage to process, then close popup
                setTimeout(() => {
                    closeTippyPopup();
                    setTimeout(callback, 500);
                }, 800);
            } else {
                addLog(`No mortgage btn for ${tileInfo.name}`, "#ff4757");
                closeTippyPopup();
                setTimeout(callback, 500);
            }
        }, 1200);
    }

    // Mortgage all owned unmortgaged properties via UI clicks, then call onDone
    function mortgageAllOnTurn(onDone) {
        if (!state.ws || !state.pid || !state.isMyTurn) {
            if (onDone) onDone();
            return;
        }

        const tiles = findOwnedUnmortgagedTiles();
        if (tiles.length === 0) {
            if (onDone) onDone();
            return;
        }

        addLog(`Mortgaging ${tiles.length} properties via UI...`, "#f39c12");
        let idx = 0;

        function next() {
            if (idx >= tiles.length || !state.isMyTurn) {
                if (onDone) onDone();
                return;
            }
            mortgageOneViaUI(tiles[idx], () => {
                idx++;
                next();
            });
        }
        next();
    }

        if (tiles.length === 0) {
            addLog("No properties to mortgage!", "#ff4757");
            return;
        }
        mortgageOneViaUI(tiles[0], () => {});
    }

    function clickSafeButtons() {
        // SAFETY: If a trade popup is open, DO NOT click any buttons
        if (isTradePopupOpen()) {
            addLog("Trade popup detected - skipping clicks", "#ff4757");
            return false;
        }

        const buttons = document.querySelectorAll('button, [role="button"]');
        for (const btn of buttons) {
            // Skip invisible/disabled buttons
            if (btn.offsetWidth === 0 || btn.offsetHeight === 0) continue;
            if (btn.disabled || btn.getAttribute('aria-disabled') === 'true') continue;
            // Skip our own dashboard buttons
            if (btn.closest('#mag-v17') || btn.closest('#mag-props-panel')) continue;

            const text = (btn.innerText || '').trim();
            const textLower = text.toLowerCase();

            // HARD BLOCK: Never click anything that could accept a trade
            if (BLOCKED_TEXTS.some(blocked => textLower.includes(blocked))) continue;

            // SAFETY: Skip if this button is inside a trade UI
            if (isInsideTradeUI(btn)) continue;

            // Check against safe patterns (always clickable)
            for (const pat of SAFE_PATTERNS) {
                if (pat.test(text)) {
                    addLog(`UI-CLICK: ${text}`, "#f39c12");
                    reactClick(btn);
                    return true;
                }

            // Check turn-specific patterns (only when it's our turn + auto-end-turn enabled + not mortgaging)
            if (state.isMyTurn && isAutoEndTurn() && state.mortgagingDone) {

                        addLog(`UI-CLICK: ${text}`, "#f39c12");
                        reactClick(btn);
                        return true;
                    }
                }
            }
        }
        return false;
    }

    // 10. REACT FIBER CLICK (for UI buttons)
    // ============================================================
    function reactClick(el) {
        const fakeEvent = { preventDefault(){}, stopPropagation(){}, target: el, currentTarget: el, nativeEvent: {} };

        // Try __reactProps$ first
        const propsKey = Object.keys(el).find(k => k.startsWith('__reactProps$'));
        if (propsKey && el[propsKey] && typeof el[propsKey].onClick === 'function') {
            el[propsKey].onClick(fakeEvent);
            return true;
        }
        // Try __reactFiber$ and walk up
        const fiberKey = Object.keys(el).find(k => k.startsWith('__reactFiber$') || k.startsWith('__reactInternalInstance$'));
        if (fiberKey) {
            let fiber = el[fiberKey];
            let depth = 0;
            while (fiber && depth < 20) {
                if (fiber.memoizedProps) {
                    if (typeof fiber.memoizedProps.onClick === 'function') {
                        fiber.memoizedProps.onClick(fakeEvent);
                        return true;
                    }
                    if (typeof fiber.memoizedProps.onPointerUp === 'function') {
                        fiber.memoizedProps.onPointerUp(fakeEvent);
                        return true;
                    }
                }
                fiber = fiber.return;
                depth++;
            }
        }
        // Last resort: native click
        el.click();
        return true;
    }

    // 11. IDENTITY SCANNER
    // ============================================================
    function scanForIdentity(obj) {
        if (state.pid || !obj || typeof obj !== 'object') return;

        // Method 1: Match by CONFIG.targetName
        const scan = (o) => {
            if (o && typeof o === 'object') {
                if (o.name && typeof o.name === 'string' &&
                    o.name.toLowerCase().includes(CONFIG.targetName.toLowerCase()) && o.id) {
                    lockIdentity(o);
                    return;
                }
                Object.values(o).forEach(v => { if (v && typeof v === 'object') scan(v); });
            }
        };
        scan(obj);

        // Method 2: If we have participants list, find the only non-bot human
        if (!state.pid && obj.gameState?.participants) {
            const humans = obj.gameState.participants.filter(p => !p.isBot);
            if (humans.length === 1) {
                lockIdentity(humans[0]);
            }
        }
        if (!state.pid && obj.participantsOrder) {
            const humans = obj.participantsOrder.filter(p => !p.isBot);
            if (humans.length === 1) {
                lockIdentity(humans[0]);
            }
        }
    }

    function lockIdentity(player) {
        if (state.pid) return;
        state.pid = player.id;
        state.myMoney = (typeof player.money === 'number') ? player.money : 1500;
        state.myPosition = player.position || 0;
        updateMeUI(player.name);
        addLog(`Identity: ${player.name} (${state.pid.slice(0,8)})`, "#2ecc71");
        updateStatsUI();
    }

    // 11b. LOBBY NET WORTH CALCULATOR (V18.1 FIX)
    // ============================================================
    // Extracted into its own function so it can be called immediately
    // when a player leaves (not just every 5s in the main loop).
    // FIX: Only counts properties whose ownerId is a CURRENT player in state.players.
    // When a player leaves/goes bankrupt, their entry is deleted from state.players
    // and their properties' ownerId is cleared — so their cash AND property value
    // both drop out of the total immediately.
    function recalcLobbyNet() {
        let total = 0;
        const activePlayerIds = new Set(Object.keys(state.players));

        // Sum all active player cash (including negative = debt)
        Object.values(state.players).forEach(p => {
            const money = p.money;
            if (typeof money === 'number' && !isNaN(money)) total += money;
        });

        // Sum property values — only for properties owned by active players
        state.blocks.forEach(b => {
            if (!b.ownerId || !b.price) return;
            // V18.1 FIX: Skip properties owned by players no longer in the game
            if (!activePlayerIds.has(b.ownerId)) return;
            // Mortgaged properties: the cash was already received (counted in player money)
            // so don't double-count. Only add unmortgaged property mortgage value.
            if (!b.isMortgaged) {
                total += Math.floor(b.price / 2);
            }
            // House/hotel improvements add value (half of what was spent)
            if (b.level && b.level > 0 && !b.isMortgaged) {
                if (b.level <= 4) {
                    total += Math.floor((b.level * (b.housePrice || 0)) / 2);
                } else if (b.level === 5) {
                    total += Math.floor((4 * (b.housePrice || 0) + (b.hotelPrice || 0)) / 2);
                }
            }
        });

        state.lobbyNet = total;

        // Update lobby net worth & 16% tax on dashboard
        const taxAmount = Math.floor(state.lobbyNet * CONFIG.tax);
        const netEl = document.getElementById('mag-net');
        const taxEl = document.getElementById('mag-tax');
        if (netEl) netEl.innerText = state.lobbyNet;
        if (taxEl) taxEl.innerText = taxAmount;

    }
